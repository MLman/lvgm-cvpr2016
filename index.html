Data synthesis

clc;clear;close all;
rng default
N = 70;
NO = 50;
NC = 10;
density = .5e-1; % a rough estimate of the amount of edges
[theta,theta_GT]= random_sparse_network(NO, NC, N, density);

% covariance matrix
sigma = pinv(theta);
sigma = (sigma+sigma')./2;
mu = zeros(1,N);

nsamples = 2000;
r = mvnrnd(mu,sigma,nsamples);
Xo = r(1:nsamples, 1:NO+NC);
Cov = Xo'*Xo./nsamples;
P = pinv(Cov);
